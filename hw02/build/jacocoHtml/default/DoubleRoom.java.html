public class DoubleRoom extends RoomItem implements HotelBookingSystem{

  protected Integer maximumOccupancy = 2;

  @Override
  public String toString() {
    return "DoubleRoom{" +
        "maximumOccupancy=" + maximumOccupancy +
        ", price=" + price +
        ", numberOfGuests=" + numberOfGuests +
        '}';
  }

  public DoubleRoom(Integer maximumOccupancy, Double price, Integer numberOfGuests)
      throws invalidPriceException, invalidOccupancyException {
    super(maximumOccupancy, price, numberOfGuests);
    if (maximumOccupancy > this.maximumOccupancy) {
      throw new invalidOccupancyException("Maximum occupancy cannot be greater than 2");
    }


  }
}
