import java.util.Objects;

public abstract class RoomItem implements HotelBookingSystem{
private  static final long serivalVErsionUID = 1L;
  protected Integer maximumOccupancy;

  protected Double price;

  protected Integer numberOfGuests;
  private static final Double MIN_PRICE = 0.0;

  public RoomItem(Integer maximumOccupancy, Double price, Integer numberOfGuests) throws invalidPriceException {
    if (price < MIN_PRICE) {
      throw new invalidPriceException("Price cannot be negative");
    }
    this.maximumOccupancy = maximumOccupancy;
    this.price = price;
    this.numberOfGuests = 0;
  }
  public Boolean isAvailable() {
    return  this.numberOfGuests == 0;

  }

  public Integer getMaximumOccupancy() {
    return this.maximumOccupancy;
  }



  public Double getPrice() {
    return this.price;
  }


  public Integer getNumberOfGuests() {
    return this.numberOfGuests;
  }



  public void bookRoom (Integer numberOfGuests) throws invalidOccupancyException {
    if (numberOfGuests > this.maximumOccupancy) {
      throw new invalidOccupancyException("Number of guests cannot be greater than maximum occupancy");
    }
    if (this.numberOfGuests != 0) {
      throw new invalidOccupancyException("Room is not available");
    }
    this.numberOfGuests = numberOfGuests;
  }

  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (!(o instanceof RoomItem)) {
      return false;
    }
    RoomItem roomItem = (RoomItem) o;
    return Objects.equals(getMaximumOccupancy(), roomItem.getMaximumOccupancy())
        && Objects.equals(getPrice(), roomItem.getPrice()) && Objects.equals(
        getNumberOfGuests(), roomItem.getNumberOfGuests());
  }

  @Override
  public int hashCode() {
    return Objects.hash(getMaximumOccupancy(), getPrice(), getNumberOfGuests());
  }

}
