public class SingleRoom extends RoomItem implements HotelBookingSystem{
  private  static final long serivalVErsionUID = 1L;
  private Integer maximumOccupancy = 1;

  public SingleRoom(Integer maximumOccupancy, Double price, Integer numberOfGuests)
      throws invalidPriceException, invalidOccupancyException {
    super(maximumOccupancy, price, numberOfGuests);
    if (maximumOccupancy > this.maximumOccupancy) {
      throw new invalidOccupancyException("Maximum occupancy cannot be greater than 1");
    }
  }

  @Override
  public String toString() {
    return "SingleRoom{" +
        "maximumOccupancy=" + maximumOccupancy +
        ", price=" + price +
        ", numberOfGuests=" + numberOfGuests +
        '}';
  }
}
